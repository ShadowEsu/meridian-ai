Point your OpenAI client at Meridian. Keep everything else.

route.py
# pip install openai — no new SDK
from openai import OpenAI

client = OpenAI(
    base_url="https://api.meridian.dev/v1",
    api_key=os.environ["MERIDIAN_KEY"],
)

resp = client.chat.completions.create(
    model="auto",            # let Meridian choose
    messages=messages,
    extra_body={"policy": "balanced"},
)

# resp.meridian → {tier, model, saved_usd}
print(resp.meridian.model)   # gpt-4.1-mini
